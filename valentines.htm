<script>

// Floating hearts generator

setInterval(() => {

    const heart = document.createElement("div");

    heart.classList.add("heart");

    heart.style.left = Math.random() * window.innerWidth + "px";

    heart.style.fontSize = (20 + Math.random() * 20) + "px";

    heart.textContent = "❤️";

    document.body.appendChild(heart);


    setTimeout(() => heart.remove(), 6000);

}, 500);


// No button text sequence

const noTexts = [

    "that's sad, please don't press no again",

    "wow really, you really don't love me",

    "this is outrageous, you must have another boyfriend",

    "JUST SAY YES ALREADY",

    "too bad you're mine, press this button one more time"

];


let noClickCount = 0;


const yesBtn = document.getElementById("yesBtn");

const noBtn = document.getElementById("noBtn");

const heart = document.getElementById("heart");

const question = document.getElementById("question");


// Handle NO button presses

noBtn.addEventListener("click", () => {

    noBtn.style.animation = "shake 0.4s";


    setTimeout(() => {

        noBtn.style.animation = "";

    }, 400);


    if (noClickCount < noTexts.length) {

        noBtn.textContent = noTexts[noClickCount];

        noClickCount++;

    } else {

        noBtn.style.display = "none";

    }

});


// Handle YES button press

yesBtn.addEventListener("click", () => {

    // Confetti effect

    for (let i = 0; i < 180; i++) {

        const confetti = document.createElement("div");

        confetti.style.position = "fixed";

        confetti.style.width = "10px";

        confetti.style.height = "10px";

        confetti.style.backgroundColor = 

            `hsl(${Math.random() * 360}, 100%, 70%)`;

        confetti.style.top = "-10px";

        confetti.style.left = Math.random() * window.innerWidth + "px";

        confetti.style.opacity = 0.9;

        confetti.style.transform = `rotate(${Math.random()*360}deg)`;

        confetti.style.transition = "top 2.2s ease-out, opacity 2.2s";


        document.body.appendChild(confetti);


        setTimeout(() => {

            confetti.style.top = window.innerHeight + "px";

            confetti.style.opacity = 0;

        }, 10);


        setTimeout(() => confetti.remove(), 2300);

    }


    // Show heart

    heart.style.display = "block";


    // Change question text

    question.textContent = "YAY! ❤️";

});

</script>
